   38 -- | Convert a base16 encoded 'Text' into a raw 'ByteString'
   39 --
   40 -- @since 2.0.0
   41 fromBase16 :: Text -> Maybe ByteString
   42 fromBase16 = eitherToMaybe . convertFromBase Base16 . T.encodeUtf8
   43 
   44 -- | Convert a raw 'ByteString' into base16 encoded 'Text'
   45 --
   46 -- @since 2.0.0
   47 base16 :: ByteString -> Text
   48 base16 = T.decodeUtf8 . convertToBase Base16
   49 
   50 -- | Convert a Bech32 encoded 'Text' into a raw 'ByteString'
   51 --
   52 -- @since 2.0.0
   53 fromBech32 :: Text -> Maybe ByteString
   54 fromBech32 txt = do
   55     (_, dp) <- either (const Nothing) Just (Bech32.decodeLenient txt)
   56     Bech32.dataPartToBytes dp
   57 
   58 -- | Convert a raw 'ByteString' into a bech32 encoded 'Text'
   59 --
   60 -- @since 2.0.0
   61 bech32 :: HumanReadablePart -> ByteString -> Text
   62 bech32 hrp bytes =
   63     Bech32.encodeLenient hrp (Bech32.dataPartFromBytes bytes)
   64 
   65 -- | Convert a base58 encoded 'Text' into a raw 'ByteString'
   66 --
   67 -- @since 2.0.0
   68 fromBase58 :: Text -> Maybe ByteString
   69 fromBase58 = decodeBase58 bitcoinAlphabet . T.encodeUtf8
   70 
   71 -- | Convert a raw 'ByteString' into a base58 encoded 'Text'
   72 base58 :: ByteString -> Text
   73 base58 = T.decodeUtf8 . encodeBase58 bitcoinAlphabet
   74 
   75 -- | Convert a base64 encoded 'Text' into a raw 'ByteString'
   76 --
   77 -- @since 2.0.0
   78 fromBase64 :: Text -> Maybe ByteString
   79 fromBase64 = eitherToMaybe . convertFromBase Base64 . T.encodeUtf8
   80 
   81 -- | Convert a raw 'ByteString' into a base64 encoded 'Text', with padding.
   82 --
   83 -- @since 2.0.0
   84 base64 :: ByteString -> Text
   85 base64 = T.decodeUtf8 . convertToBase Base64
