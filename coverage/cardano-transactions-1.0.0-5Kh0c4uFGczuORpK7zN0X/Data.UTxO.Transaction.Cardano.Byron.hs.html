  103 -- | Construct a payment 'Input' for /Byron/ from primitive types.
  104 --
  105 -- __example__:
  106 --
  107 -- >>> mkInput 14 =<< fromBase16 "3b402651...aad1c0b7"
  108 -- Just (Input ...)
  109 --
  110 -- @since 1.0.0
  111 mkInput
  112     :: Word32
  113         -- ^ Input index.
  114     -> ByteString
  115         -- ^ Input transaction id. See also: 'fromBase16'.
  116     -> Maybe (Input Byron)
  117 mkInput ix bytes =
  118     case digestFromByteString @Blake2b_256 bytes of
  119         Just txId -> Just $ TxInUtxo (abstractHashFromDigest txId) ix
  120         Nothing -> Nothing
  121 
  122 -- | Construct a payment 'Output' for /Byron/ from primitive types.
  123 --
  124 -- __example__:
  125 --
  126 -- >>> mkOutput 42 =<< fromBase58 "Ae2tdPwU...DnXy319f"
  127 -- Just (Output ...)
  128 --
  129 -- @since 1.0.0
  130 mkOutput
  131     :: Natural
  132         -- ^ Output value, in Lovelace (1 Ada = 1e6 Lovelace).
  133     -> ByteString
  134         -- ^ Output Address. See also: 'fromBase58'.
  135     -> Maybe (Output Byron)
  136 mkOutput n bytes =
  137     case (fromCBOR' bytes, mkLovelace (fromIntegral n)) of
  138         (Right addr, Right coin) -> Just $ TxOut addr coin
  139         _ -> Nothing
  140   where
  141     fromCBOR' = fmap snd . deserialiseFromBytes fromCBOR .  BL.fromStrict
  142 
  143 -- | Construct a 'SignKey' for /Byron/ from primitive types.
  144 --
  145 -- __example__:
  146 --
  147 -- >>> mkSignKey =<< fromBase16 "3b402651...aad1c0b7"
  148 -- Just (SignKey ...)
  149 --
  150 -- @since 1.0.0
  151 mkSignKey
  152     :: ByteString
  153         -- ^ A extended address private key and its chain code.
  154         -- The key __must be 96 bytes__ long, internally made of two concatenated parts:
  155         --
  156         -- @
  157         -- BYTES = PRV | CC
  158         -- PRV   = 64OCTET  # a 64 bytes Ed25519 extended private key
  159         -- CC    = 32OCTET  # a 32 bytes chain code
  160         -- @
  161         --
  162         -- See also: 'fromBase16'.
  163     -> Maybe (SignKey Byron)
  164 mkSignKey = fmap SigningKey . xprvFromBytes

  170 -- Type-level constructor capturing types for 'Byron'.
  171 data Byron
  172 
  173 instance MkPayment Byron where
  174     type Init Byron = ProtocolMagicId
  175 
  176     type Input   Byron = TxIn
  177     type Output  Byron = TxOut
  178     type SignKey Byron = SigningKey
  179 
  180     type CoinSel Byron =
  181         (ProtocolMagicId, [TxIn], [TxOut])
  182 
  183     type Tx Byron = Either
  184         ErrMkPayment
  185         (ProtocolMagicId, NonEmpty TxIn, NonEmpty TxOut, TxSigData, [TxInWitness])
  186 
  187     empty :: ProtocolMagicId -> CoinSel Byron
  188     empty pm = (pm, mempty, mempty)
  189 
  190     addInput :: TxIn -> CoinSel Byron -> CoinSel Byron
  191     addInput inp (pm, inps, outs) = (pm, inp : inps, outs)
  192 
  193     addOutput :: TxOut -> CoinSel Byron -> CoinSel Byron
  194     addOutput out (pm, inps, outs) = (pm, inps, out : outs)
  195 
  196     lock :: CoinSel Byron -> Tx Byron
  197     lock (_pm, [], _outs) = Left MissingInput
  198     lock (_pm, _inps, []) = Left MissingOutput
  199     lock (pm, inps, outs) =
  200         Right (pm, neInps, neOuts, sigData, mempty)
  201       where
  202         sigData = TxSigData $ serializeCborHash $ CC.UnsafeTx neInps neOuts (mkAttributes ())
  203         neInps  = NE.fromList $ reverse inps
  204         neOuts  = NE.fromList $ reverse outs
  205 
  206     signWith :: SigningKey -> Tx Byron -> Tx Byron
  207     signWith _ (Left e) = Left e
  208     signWith (SigningKey prv) (Right (pm, inps, outs, sigData, wits)) =
  209         Right (pm, inps, outs, sigData, VKWitness vk sig : wits)
  210       where
  211         vk :: VerificationKey
  212         vk = VerificationKey (toXPub prv)
  213 
  214         sig :: Signature TxSigData
  215         sig = CC.sign pm SignTx (SigningKey prv) sigData
  216 
  217     serialize :: Tx Byron -> Either ErrMkPayment ByteString
  218     serialize (Left e) = Left e
  219     serialize (Right (_pm, inps, outs, _sigData, wits))
  220         | NE.length inps /= length wits = Left MissingSignature
  221         | otherwise = Right $ CBOR.toStrictByteString $ toCBOR $ mkTxAux
  222             (CC.UnsafeTx inps outs (mkAttributes ()))
  223             (fromList $ reverse wits)

  226 -- Internal
  227 --
  228 -- For running the Payment DSL as a command-line, we need to be able to produce
  229 -- a text output representing the internal state.
  230 --
  231 -- There's no point about obfuscating this more than necessary; this should
  232 -- remain mainly invisible to users if used properly through pipes. Yet, it can
  233 -- be useful to have a human-friendly representation that is base16 or base64
  234 -- encoded. Three possible obvious choice:
  235 --
  236 -- - CBOR, since most data above can already be serialized to CBOR
  237 -- - JSON, instances exists on most type and are derived generically from CBOR
  238 -- - Show, although here most types don't have a corresponding 'Read' instance.
  239 --
  240 -- Since we also need to decode an encoded state, CBOR will be path of least
  241 -- resistance.
  242 
  243 -- __Internal__: Encode a 'CoinSel Byron' to CBOR.
  244 encodeCoinSel :: CoinSel Byron -> CBOR.Encoding
  245 encodeCoinSel (pm, inps, outs) = mconcat
  246     [ toCBOR pm
  247     , CBOR.encodeListLenIndef
  248     , mconcat (toCBOR <$> inps)
  249     , CBOR.encodeBreak
  250     , CBOR.encodeListLenIndef
  251     , mconcat (toCBOR <$> outs)
  252     , CBOR.encodeBreak
  253     ]
  254 
  255 -- __Internal__: Decode a 'CoinSel Byron' from CBOR.
  256 decodeCoinSel :: CBOR.Decoder s (CoinSel Byron)
  257 decodeCoinSel = (,,)
  258     <$> fromCBOR
  259     <*> decodeListIndef fromCBOR
  260     <*> decodeListIndef fromCBOR
  261 
  262 -- __Internal__: Encode a 'Tx Byron' to CBOR.
  263 encodeTx :: Tx Byron -> CBOR.Encoding
  264 encodeTx (Left e) = mconcat
  265     [ CBOR.encodeWord8 0
  266     , CBOR.encodeString $ T.pack $ show e
  267     ]
  268 encodeTx (Right (pm, inps, outs, sigData, wits)) = mconcat
  269     [ CBOR.encodeWord8 1
  270     , encodeCoinSel (pm, NE.toList inps, NE.toList outs)
  271     , toCBOR sigData
  272     , CBOR.encodeListLenIndef
  273     , mconcat (toCBOR <$> wits)
  274     , CBOR.encodeBreak
  275     ]
  276 
  277 {-# HLINT ignore decodeTx "Redundant fmap" #-}
  278 -- __Internal__: Decode a 'Tx Byron' from CBOR.
  279 decodeTx :: CBOR.Decoder s (Tx Byron)
  280 decodeTx = do
  281     CBOR.decodeWord8 >>= \case
  282         0 -> fmap T.unpack CBOR.decodeString >>= \case
  283             str | str == show MissingInput     -> pure $ Left MissingInput
  284             str | str == show MissingOutput    -> pure $ Left MissingOutput
  285             str | str == show MissingSignature -> pure $ Left MissingSignature
  286             _ -> fail $
  287                 "Invalid error constructor found in 'Tx Byron'. Constructor must \
  288                 \be one of: " <> unwords (show <$>
  289                     [ MissingInput, MissingOutput, MissingSignature])
  290         1 -> do
  291             (pm, inps, outs) <- decodeCoinSel
  292             sigData <- fromCBOR
  293             wits <- decodeListIndef fromCBOR
  294             case (nonEmpty inps, nonEmpty outs) of
  295                 (Nothing, _) -> fail
  296                     "Empty list of inputs found in 'Tx Byron'. This is impossible \
  297                     \unless the data has been modified by hand."
  298                 (_, Nothing) -> fail
  299                     "Empty list of outputs found in 'Tx Byron'. This is impossible \
  300                     \unless the data has been modified by hand."
  301                 (Just neInps, Just neOuts) ->
  302                     pure $ Right (pm, neInps, neOuts, sigData, wits)
  303 
  304         _ -> fail
  305             "'Tx Byron' has been modified  with and is now invalid. The first \
  306             \byte must be either 0 or 1, followed by respectively an error \
  307             \constructor or constituants of the intermediate transaction."
  308 
  309 -- __Internal__ Decode an arbitrary long list 'sandwiched' by markers.
  310 decodeListIndef :: forall s a. CBOR.Decoder s a -> CBOR.Decoder s [a]
  311 decodeListIndef decodeOne = do
  312     _ <- CBOR.decodeListLenIndef
  313     CBOR.decodeSequenceLenIndef (flip (:)) [] reverse decodeOne
