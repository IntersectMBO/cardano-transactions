   23 -- $overview
   24 --
   25 -- This module offers an abstraction over the construction of simple UTxO
   26 -- transactions. Such a transaction is called a /Payment/. Payments are
   27 -- constructed by selecting inputs and outputs, and signing each input
   28 -- independently.
   29 --
   30 -- This can be represented as a state machine, such as:
   31 --
   32 -- @
   33 --                          empty
   34 --                            |
   35 --      *-----------------*   |   *----------------*
   36 --      |                 |   |   |                |
   37 --      |                 v   v   v                |
   38 --      *--- addOutput ---=========--- addInput ---*
   39 --                            |
   40 --                            |
   41 --                          lock  *----------------*
   42 --                            |   |                |
   43 --                            |   v                |
   44 --                        =========--- signWith ---*
   45 --                            |
   46 --                            |
   47 --                        serialize
   48 --                            |
   49 --                            |
   50 --                            v
   51 --                           ...
   52 -- @
   53 --
   54 -- Functions from the 'MkPayment' interfaces are designed to be easy to chain
   55 -- like pipes. To give a concrete example, let's see how to construct a
   56 -- transaction with /one input/ and /two outputs/:
   57 --
   58 -- @
   59 -- import Data.Function
   60 --     ( (&) )
   61 -- import qualified Data.UTxO.Transaction as Tx
   62 --
   63 -- bytes = Tx.empty
   64 --       & Tx.addInput  input0
   65 --       & Tx.addOutput output0
   66 --       & Tx.addOutput output1
   67 --       & Tx.lock
   68 --       & Tx.signWith  key0
   69 --       & Tx.serialize
   70 -- @
   71 --
   72 -- The reason why the 'lock' step is required is because, in order to create
   73 -- witnesses, one must produce signatures of the whole transaction body (inputs
   74 -- + outputs). Hence, signatures can only be added once all inputs and all
   75 -- outputs have already been addded.
   76 --
   77 -- This is correct-by-construction thanks to the 'lock' step and the distinction
   78 -- between 'CoinSel' and 'Tx'.
   79 
   80 -- | Abstract interface for constructing a /Payment/.
   81 class MkPayment dom where
   82     -- | Some initial extra data needed to construct and or serialize the
   83     -- transaction. This is a free-type which is implementation-specific.
   84     --
   85     -- For example, Byron transactions requires the underlying protocol magic in
   86     -- order to produce witnesses. This is an appropriate place for.
   87     type Init    dom :: *
   88 
   89     -- | Type representation of transaction 'Input'
   90     type Input   dom = inp | inp -> dom
   91 
   92     -- | Type representation of transaction 'Output'
   93     type Output  dom = out | out -> dom
   94 
   95     -- | Type representation of transaction 'SignKey'
   96     type SignKey dom = key | key -> dom
   97 
   98     -- | An intermediate type for representing a transaction under construction.
   99     -- 'Input' and 'Output' can still be added to a 'CoinSel' and must be added
  100     -- before trying to sign any of the inputs.
  101     type CoinSel dom = sel | sel -> dom
  102 
  103     -- | A transaction is a locked 'CoinSel' to which it is no longer possible
  104     -- to add new 'Input' and or 'Output'. Only 'Tx' can be serialized.
  105     type Tx      dom = tx  | tx  -> dom
  106 
  107     -- * Transaction construction
  108     empty     :: Init dom -> CoinSel dom
  109     addInput  :: Input  dom -> CoinSel dom -> CoinSel dom
  110     addOutput :: Output dom -> CoinSel dom -> CoinSel dom
  111 
  112     -- * Transaction signature
  113     lock      :: CoinSel dom -> Tx dom
  114     signWith  :: SignKey dom -> Tx dom -> Tx dom
  115 
  116     -- * Transaction serialization
  117     serialize :: Tx dom -> Either ErrMkPayment ByteString
  118 
  119 -- | Possible errors occuring when constructing a /Payment/. This can in
  120 -- practice be safely ignored if the transaction is known to be well-formed.
  121 data ErrMkPayment
  122     = MissingInput
  123         -- ^ Payments must have at least one input.
  124     | MissingOutput
  125         -- ^ Payments must have at least one output
  126     | MissingSignature
  127         -- ^ Payments must have a signature for each input.
  128     deriving (Show, Read, Eq)
  129 
  130 -- $note
  131 --
  132 -- It is relatively easy to tweak the interface below to get it fully type-safe
  133 -- by using an inductive type, such as:
  134 --
  135 -- @
  136 -- data N = Zero | Some N
  137 --
  138 -- class MkPayment dom where
  139 --     type Init    dom :: *
  140 --
  141 --     type Input   dom = inp | inp -> dom
  142 --     type Output  dom = out | out -> dom
  143 --     type SignKey dom = key | key -> dom
  144 --
  145 --     type CoinSel dom = (sel :: N -> N -> *) | sel -> dom
  146 --     type Tx      dom = (tx  :: N -> *)      | tx  -> dom
  147 --
  148 --     empty     :: Init dom -> CoinSel dom Zero Zero
  149 --
  150 --     addInput  :: Input  dom -> CoinSel dom n m -> CoinSel dom (Some n) m
  151 --     addOutput :: Output dom -> CoinSel dom n m -> CoinSel dom n (Some m)
  152 --
  153 --     lock      :: CoinSel dom (Some n) (Some m) -> Tx dom (Some n)
  154 --
  155 --     signWith :: SignKey dom -> Tx dom (Some n) -> Tx dom n
  156 --
  157 --     serialize :: Tx dom Zero -> ByteString
  158 -- @
  159 --
  160 -- Such approach is great for ensuring that the transaction can't be ill-formed
  161 -- and works well when manually constructing the transaction as shown above.
  162 --
  163 -- It gets however quite complex when one wants to start folding or mapping on a
  164 -- list of inputs or outputs where their numbers isn't known at compile-time!
  165 -- Indeed, each call to 'addInput' does actually return a new type!
  166 --
  167 -- Let's say we want to write the following function:
  168 --
  169 -- @
  170 -- addMany :: [Input dom] -> CoinSel dom n m -> CoinSel dom ? m
  171 -- @
  172 --
  173 -- What should be the type of `?`. This can't possibly known at compile-time.
