  118 -- | Construct a payment 'Input' for /Byron/ from primitive types.
  119 --
  120 -- __example__:
  121 --
  122 -- >>> mkInput 14 =<< fromBase16 "3b402651...aad1c0b7"
  123 -- Just (Input ...)
  124 --
  125 -- @since 1.0.0
  126 mkInput
  127     :: Word32
  128         -- ^ Input index.
  129     -> ByteString
  130         -- ^ Input transaction id. See also: 'fromBase16'.
  131     -> Maybe (Input Byron)
  132 mkInput ix bytes =
  133     case digestFromByteString @Blake2b_256 bytes of
  134         Just txId -> Just $ TxInUtxo (AbstractHash txId) ix
  135         Nothing -> Nothing
  136 
  137 -- | Construct a payment 'Output' for /Byron/ from primitive types.
  138 --
  139 -- __example__:
  140 --
  141 -- >>> mkOutput 42 =<< fromBase58 "Ae2tdPwU...DnXy319f"
  142 -- Just (Output ...)
  143 --
  144 -- @since 1.0.0
  145 mkOutput
  146     :: Natural
  147         -- ^ Output value, in Lovelace (1 Ada = 1e6 Lovelace).
  148     -> ByteString
  149         -- ^ Output Address. See also: 'fromBase58'.
  150     -> Maybe (Output Byron)
  151 mkOutput n bytes =
  152     case (fromCBOR' bytes, mkLovelace (fromIntegral n)) of
  153         (Right addr, Right coin) -> Just $ TxOut addr coin
  154         _ -> Nothing
  155   where
  156     fromCBOR' = fmap snd . deserialiseFromBytes fromCBOR .  BL.fromStrict
  157 
  158 -- | Construct a 'SignKey' for /Byron/ from primitive types.
  159 --
  160 -- __example__:
  161 --
  162 -- >>> mkSignKey =<< fromBase16 "3b402651...aad1c0b7"
  163 -- Just (SignKey ...)
  164 --
  165 -- @since 1.0.0
  166 mkSignKey
  167     :: ByteString
  168         -- ^ A extended address private key and its chain code.
  169         -- The key __must be 96 bytes__ long, internally made of two concatenated parts:
  170         --
  171         -- @
  172         -- BYTES = PRV | CC
  173         -- PRV   = 64OCTET  # a 64 bytes Ed25519 extended private key
  174         -- CC    = 32OCTET  # a 32 bytes chain code
  175         -- @
  176         --
  177         -- See also: 'fromBase16'.
  178     -> Maybe (SignKey Byron)
  179 mkSignKey bytes
  180     | BS.length bytes /= 96 = Nothing
  181     | otherwise = do
  182         let (prv, cc) = BS.splitAt 64 bytes
  183         pub <- ed25519ScalarMult (BS.take 32 prv)
  184         fmap SigningKey $ eitherToMaybe $ xprv $ prv <> pub <> cc
  185   where
  186     ed25519ScalarMult :: ByteString -> Maybe ByteString
  187     ed25519ScalarMult bs = do
  188         scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
  189         pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar

  215 -- MkPayment instance

  218 -- Type-level constructor capturing types for 'Byron'.
  219 data Byron
  220 
  221 instance MkPayment Byron where
  222     type Init Byron = ProtocolMagicId
  223 
  224     type Input   Byron = TxIn
  225     type Output  Byron = TxOut
  226     type SignKey Byron = SigningKey
  227 
  228     type CoinSel Byron =
  229         (ProtocolMagicId, [TxIn], [TxOut])
  230 
  231     type Tx Byron = Either
  232         ErrMkPayment
  233         (ProtocolMagicId, NonEmpty TxIn, NonEmpty TxOut, TxSigData, [TxInWitness])
  234 
  235     empty :: ProtocolMagicId -> CoinSel Byron
  236     empty pm = (pm, mempty, mempty)
  237 
  238     addInput :: TxIn -> CoinSel Byron -> CoinSel Byron
  239     addInput inp (pm, inps, outs) = (pm, inp : inps, outs)
  240 
  241     addOutput :: TxOut -> CoinSel Byron -> CoinSel Byron
  242     addOutput out (pm, inps, outs) = (pm, inps, out : outs)
  243 
  244     lock :: CoinSel Byron -> Tx Byron
  245     lock (_pm, [], _outs) = Left MissingInput
  246     lock (_pm, _inps, []) = Left MissingOutput
  247     lock (pm, inps, outs) =
  248         Right (pm, neInps, neOuts, sigData, mempty)
  249       where
  250         sigData = TxSigData $ hash $ CC.UnsafeTx neInps neOuts (mkAttributes ())
  251         neInps  = NE.fromList $ reverse inps
  252         neOuts  = NE.fromList $ reverse outs
  253 
  254     signWith :: SigningKey -> Tx Byron -> Tx Byron
  255     signWith _ (Left e) = Left e
  256     signWith (SigningKey prv) (Right (pm, inps, outs, sigData, wits)) =
  257         Right (pm, inps, outs, sigData, VKWitness vk sig : wits)
  258       where
  259         vk :: VerificationKey
  260         vk = VerificationKey (toXPub prv)
  261 
  262         sig :: Signature TxSigData
  263         sig = CC.sign pm SignTx (SigningKey prv) sigData
  264 
  265     serialize :: Tx Byron -> Either ErrMkPayment ByteString
  266     serialize (Left e) = Left e
  267     serialize (Right (_pm, inps, outs, _sigData, wits))
  268         | NE.length inps /= length wits = Left MissingSignature
  269         | otherwise = Right $ CBOR.toStrictByteString $ toCBOR $ mkTxAux
  270             (CC.UnsafeTx inps outs (mkAttributes ()))
  271             (fromList $ reverse wits)

  274 -- Internal
  275 --
  276 -- For running the Payment DSL as a command-line, we need to be able to produce
  277 -- a text output representing the internal state.
  278 --
  279 -- There's no point about obfuscating this more than necessary; this should
  280 -- remain mainly invisible to users if used properly through pipes. Yet, it can
  281 -- be useful to have a human-friendly representation that is base16 or base64
  282 -- encoded. Three possible obvious choice:
  283 --
  284 -- - CBOR, since most data above can already be serialized to CBOR
  285 -- - JSON, instances exists on most type and are derived generically from CBOR
  286 -- - Show, although here most types don't have a corresponding 'Read' instance.
  287 --
  288 -- Since we also need to decode an encoded state, CBOR will be path of least
  289 -- resistance.
  290 
  291 -- __Internal__: Encode a 'CoinSel Byron' to CBOR.
  292 encodeCoinSel :: CoinSel Byron -> CBOR.Encoding
  293 encodeCoinSel (pm, inps, outs) = mconcat
  294     [ toCBOR pm
  295     , CBOR.encodeListLenIndef
  296     , mconcat (toCBOR <$> inps)
  297     , CBOR.encodeBreak
  298     , CBOR.encodeListLenIndef
  299     , mconcat (toCBOR <$> outs)
  300     , CBOR.encodeBreak
  301     ]
  302 
  303 -- __Internal__: Decode a 'CoinSel Byron' from CBOR.
  304 decodeCoinSel :: CBOR.Decoder s (CoinSel Byron)
  305 decodeCoinSel = (,,)
  306     <$> fromCBOR
  307     <*> decodeListIndef fromCBOR
  308     <*> decodeListIndef fromCBOR
  309 
  310 -- __Internal__: Encode a 'Tx Byron' to CBOR.
  311 encodeTx :: Tx Byron -> CBOR.Encoding
  312 encodeTx (Left e) = mconcat
  313     [ CBOR.encodeWord8 0
  314     , CBOR.encodeString $ T.pack $ show e
  315     ]
  316 encodeTx (Right (pm, inps, outs, sigData, wits)) = mconcat
  317     [ CBOR.encodeWord8 1
  318     , encodeCoinSel (pm, NE.toList inps, NE.toList outs)
  319     , toCBOR sigData
  320     , CBOR.encodeListLenIndef
  321     , mconcat (toCBOR <$> wits)
  322     , CBOR.encodeBreak
  323     ]
  324 
  325 -- __Internal__: Decode a 'Tx Byron' from CBOR.
  326 decodeTx :: CBOR.Decoder s (Tx Byron)
  327 decodeTx = do
  328     CBOR.decodeWord8 >>= \case
  329         0 -> fmap T.unpack CBOR.decodeString >>= \case
  330             str | str == show MissingInput     -> pure $ Left MissingInput
  331             str | str == show MissingOutput    -> pure $ Left MissingOutput
  332             str | str == show MissingSignature -> pure $ Left MissingSignature
  333             _ -> fail $
  334                 "Invalid error constructor found in 'Tx Byron'. Constructor must \
  335                 \be one of: " <> unwords (show <$>
  336                     [ MissingInput, MissingOutput, MissingSignature])
  337         1 -> do
  338             (pm, inps, outs) <- decodeCoinSel
  339             sigData <- fromCBOR
  340             wits <- decodeListIndef fromCBOR
  341             case (nonEmpty inps, nonEmpty outs) of
  342                 (Nothing, _) -> fail
  343                     "Empty list of inputs found in 'Tx Byron'. This is impossible \
  344                     \unless the data has been modified by hand."
  345                 (_, Nothing) -> fail
  346                     "Empty list of outputs found in 'Tx Byron'. This is impossible \
  347                     \unless the data has been modified by hand."
  348                 (Just neInps, Just neOuts) ->
  349                     pure $ Right (pm, neInps, neOuts, sigData, wits)
  350 
  351         _ -> fail
  352             "'Tx Byron' has been modified  with and is now invalid. The first \
  353             \byte must be either 0 or 1, followed by respectively an error \
  354             \constructor or constituants of the intermediate transaction."
  355 
  356 -- __Internal__ Decode an arbitrary long list 'sandwiched' by markers.
  357 decodeListIndef :: forall s a. CBOR.Decoder s a -> CBOR.Decoder s [a]
  358 decodeListIndef decodeOne = do
  359     _ <- CBOR.decodeListLenIndef
  360     CBOR.decodeSequenceLenIndef (flip (:)) [] reverse decodeOne
