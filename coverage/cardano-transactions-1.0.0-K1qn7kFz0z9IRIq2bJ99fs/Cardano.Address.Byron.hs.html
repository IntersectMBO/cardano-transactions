   39 -- | A type isomorphic to an 'ByteString' to represent addresses.
   40 newtype Address = Address
   41     { unAddress :: ByteString }
   42     deriving (Show)
   43 
   44 -- | Re-construct address attributes from an address
   45 getAddrAttributes :: Address -> Maybe AddrAttributes
   46 getAddrAttributes addr = do
   47     attributes <- deserializeAttributes addr
   48     let networkMagic = findNetworkMagic attributes
   49     let hdPayload = findHDPayload attributes
   50     pure $ AddrAttributes hdPayload networkMagic
   51 
   52 -- | Extract an HD payload from a list of raw address attributes.
   53 findHDPayload :: [(Word8, ByteString)] -> Maybe HDAddressPayload
   54 findHDPayload attributes =
   55     case filter (\(tag,_) -> tag == 1) attributes of
   56         [(1, bytes)] ->
   57             HDAddressPayload <$> decodeNestedBytes CBOR.decodeBytes bytes
   58         _ ->
   59             Nothing
   60 
   61 -- | Extract the network magic from a list of raw address attributes.
   62 --
   63 -- NOTE: the attribute is only present for 'testnet' addresses, so the absence
   64 -- of attributes indicates a 'Mainnet' or 'Staging' address.
   65 findNetworkMagic :: HasCallStack => [(Word8, ByteString)] -> NetworkMagic
   66 findNetworkMagic attributes =
   67     case filter (\(tag, _) -> tag == 2) attributes of
   68         [(2, bytes)] ->
   69             case deserialiseCbor CBOR.decodeInt32 bytes of
   70                 Nothing -> error "malformed network magic in address?"
   71                 Just pm -> NetworkTestnet $ fromIntegral pm
   72         _ ->
   73             NetworkMainOrStage
   74 
   75 -- | Extract the HD Payload from a set of raw attributes
   76 deserializeAttributes :: Address -> Maybe [(Word8, ByteString)]
   77 deserializeAttributes (Address addr) = do
   78     payload <- deserialiseCbor decodeAddressPayload addr
   79     deserialiseCbor decodeAllAttributes' payload
   80   where
   81     decodeAllAttributes' = do
   82         _ <- CBOR.decodeListLenCanonicalOf 3
   83         _ <- CBOR.decodeBytes
   84         decodeAllAttributes
   85 
   86     decodeAllAttributes = do
   87         n <- CBOR.decodeMapLenCanonical -- Address Attributes length
   88         replicateM n decodeAttr
   89       where
   90         decodeAttr = (,) <$> CBOR.decodeWord8 <*> CBOR.decodeBytes
   91 
   92     decodeAddressPayload = do
   93         _ <- CBOR.decodeListLenCanonicalOf 2
   94         _ <- CBOR.decodeTag
   95         bytes <- CBOR.decodeBytes
   96         _ <- CBOR.decodeWord32 -- CRC
   97         return bytes

  103 decodeNestedBytes
  104     :: MonadFail m
  105     => (forall s. CBOR.Decoder s r)
  106     -> ByteString
  107     -> m r
  108 decodeNestedBytes dec bytes =
  109     case CBOR.deserialiseFromBytes dec (BL.fromStrict bytes) of
  110         Right ("", res) ->
  111             pure res
  112         Right _ ->
  113             fail "Leftovers when decoding nested bytes"
  114         _ ->
  115             fail "Could not decode nested bytes"
  116 
  117 deserialiseCbor
  118     :: (forall s. CBOR.Decoder s a)
  119     -> ByteString
  120     -> Maybe a
  121 deserialiseCbor dec =
  122     fmap snd . eitherToMaybe . CBOR.deserialiseFromBytes dec . BL.fromStrict
