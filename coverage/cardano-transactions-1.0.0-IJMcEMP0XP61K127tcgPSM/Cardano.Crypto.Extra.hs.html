   23 -- | Convert a raw 96-byte 'ByteString' into a @cardano-crypto@ 'XPrv'
   24 --
   25 -- In memory, XPrv are basically a concatenation of three parts:
   26 --
   27 --     (encrypted) private key | public key | chain code
   28 --     <----------------------> <----------> <---------->
   29 --             64 bytes           32 bytes     32 bytes
   30 --
   31 -- NOTE 1:
   32 --   The input bytestring is expected to be made of only the encrypted private
   33 --   part and the chain code, while the public part is re-calculated from the
   34 --   private part.
   35 --
   36 -- NOTE 2:
   37 --   This module, we do not encrypt the private part in-memory, but it could be
   38 --   in practice.
   39 xprvFromBytes :: ByteString -> Maybe XPrv
   40 xprvFromBytes bytes
   41     | BS.length bytes /= 96 = Nothing
   42     | otherwise = do
   43         let (prv, cc) = BS.splitAt 64 bytes
   44         pub <- ed25519ScalarMult (BS.take 32 prv)
   45         eitherToMaybe $ xprv $ prv <> pub <> cc
   46   where
   47       ed25519ScalarMult :: ByteString -> Maybe ByteString
   48       ed25519ScalarMult bs = do
   49           scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
   50           pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar
